179 __inline static U32 rt_inc_qi (U32 size, U8 *count, U8 *first) {
180  U32 cnt,c2;
181 #ifdef __USE_EXCLUSIVE_ACCESS
182  do {
183  if ((cnt = __ldrex(count)) == size) {
184  __clrex();
185  return (cnt); }
186  } while (__strex(cnt+1, count));
187  do {
188  c2 = (cnt = __ldrex(first)) + 1;
189  if (c2 == size) c2 = 0;
190  } while (__strex(c2, first));
191 #else
192  __disable_irq();
193  if ((cnt = *count) < size) {
194  *count = cnt+1;
195  c2 = (cnt = *first) + 1;
196  if (c2 == size) c2 = 0;
197  *first = c2;
198  }
199  __enable_irq ();
200 #endif
201  return (cnt);
202 }

204 __inline static void rt_tmr_init (void) {
205 #if !(__TARGET_ARCH_6S_M)
206  int sh,prigroup;
207 #endif
208  NVIC_ST_RELOAD = os_trv;
209  NVIC_ST_CURRENT= 0;
210  NVIC_ST_CTRL = 0x0007;
211  NVIC_SYS_PRI3 |= 0xFFFF0000;
212 #if (__TARGET_ARCH_6S_M)
213  NVIC_SYS_PRI2 |= (NVIC_SYS_PRI3<<1) & 0xFC000000;
214 #else
215  sh = 8 - __clz (~(NVIC_SYS_PRI3 & 0xFF000000));
216  prigroup = ((NVIC_AIR_CTRL >> 8) & 0x07);
217  if (prigroup >= sh) {
218  sh = prigroup + 1;
219  }
220  NVIC_SYS_PRI2 = ((0xFEFFFFFF << sh) & 0xFF000000) | (NVIC_SYS_PRI2 & 0x00FFFFFF);
221 #endif
222 }
